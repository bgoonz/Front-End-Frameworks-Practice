    <script src="https://ajax.googleapis.com/ajax/libs/jquery/3.3.1/jquery.min.js"></script>
    <script>
        var score = 0;
        var gameover = false;
        
        $(function(){
           $('#start').click(startGame); 
        });
        
        function startGame(){
            //make game board
            score=0;
            $('.score').text(score);
            // start gameplay
            gameover = false;
            setTimeout(function(){
                return gameend();
            },5000);
        }
        
        function gameend(){
            gameover = true;
            $('.message').html('GAME OVER');
        }
        
        
        
        
        
        
        
    </script>
